2.指令

#安装hugo
brew install hugo

#新建站点
hugo new site xx

#添加主题
cd xx
git clone https://github.com/rmsubekti/the-roots-home.git themes/the-roots-home
hugo server -t the-roots-home --buildDrafts

#打包
hugo --theme=the-roots-home --baseUrl="https://Apollo2light.github.io" --buildDrafts

4.支持视频

//直接更改主题，在主题模板里更改

//1.single20行添加video标签
<video id="my_video" style="height: 300px;width:100%" controls="controls"></video>

//2.动态添加视频源
<script>
    try{
        //获得md中写的视频链接
        let src = document.getElementsByTagName('code')[0].innerText;
        //如果该链接的内容是视频mp4格式就添加video标签
        if(src.indexOf("mp4")!=-1){
            document.getElementById('my_video').setAttribute("src",src)
        }else{
            //如果该链接的内容不是视频格式就删除video标签
            let video = document.getElementById('my_video');
            if (video != null){
                video.parentNode.removeChild(video);
            }
        }
    }catch(e){
        //如果该链接的内容不是视频格式就删除video标签
        let video = document.getElementById('my_video');
        if (video != null){
            video.parentNode.removeChild(video);
        }
    }
</script>
