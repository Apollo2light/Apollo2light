let data = {
        name:'a',
        address:'b'
    }

    const obs = new Observer(data);
    console.log(obs);

    let vm = {}
    vm._data = data = obs;

    function Observer(obj){
        const keys = Object.keys(obj);
        keys.forEach((k) => {
            Object.defineProperty(this,k,{
                get(){
                    return obj[k];
                },
                set(val){
                    console.log(`${k}被改了,需要解析模版，生成虚拟dom，渲染页面`)
                    obj[k] = val;
                }
            })
        })
    }

    new Vue({
        el:'#root',
        data:{
            firstname:'a',
            lastname:'b'
        },
        //计算属性,处理同步属性变化
        computed:{
            'fullname':{
                //初次使用或依赖属性发生变化时get自动执行，其余时刻使用缓存
                get(){
                    return this.firstname + this.lastname;
                },
                //fullname值改变时被调用
                set(value){
                    //value就是改变的值的内容
                    console.log(value)
                }
            }
        }

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <script src="./vue2.js"></script>
    <title>vue2</title>

<body>
    
    <div id="root">
        <div>监听属性watch</div><br>
        姓：<input type="text" v-model="firstname"><br><br>
        名：<input type="text" v-model="lastname"><br><br>
        全名：<span>{{fullname}}</span>
        <hr>
    </div>

    <script>
    new Vue({
        el:'#root',
        data:{
            firstname:'a',
            lastname:'b',
            fullname:'a-b'
        },
        //监听属性，处理异步属性变化
        watch:{
            'firstname':{
                //深度监视多级结构中所有变量的变化
                deep:true,
                //初始时就调用1次
                immediate:true,
                //firstName属性改变时调用
                handler(newValue,oldValue){
                    setTimeout(()=>{
                        this.fullname = newValue + '-' +this.lastname;
                    },1000)
                }
            },
            lastname(newValue){
                this.fullname = this.firstname + '-' + newValue;
            }
        }
    })
    </script>
</body>
</html>

    <div id="root">
        <hr>
        <!-- class绑定的类名不确定 -->
        <div class="basic" :class="color" @click="change">{{name}}</div>

        <hr>
        <!-- class绑定的样式个数和名字不确定 -->
        <div class="basic" :class="arr">{{name}}</div>
        <button @click="change2">去除</button>

        <hr>
        <!-- class绑定的样式个数和名字确定, 但要动态决定用不用 -->
        <div class="basic" :class="obj">{{name}}</div>
        <button @click="change3">切换</button>

        <hr>
        <!-- style样式 -->
        <div class="basic" :style="obj1">{{name}}</div>


    </div>
    <script>

    let vm = new Vue({
        el:'#root',
        data(){
            return{
                name:'kk',
                color:'basic',
                arr:['a','b','c'],
                obj:{
                    'a':false,
                    'b':false
                },
                obj1:{
                    fontSize: '80px'
                }
            }
        },
        methods:{
            change(){
                const arr = ['basic','normal','special'];
                let index = Math.floor(Math.random()*3);
                this.color = arr[index];
            },
            change2(){
                this.arr.shift();
            },
            change3(){
                let num = Math.random()*3;
                if(num>1){
                    this.obj.a = true;
                    this.obj.b = true;
                }else{
                    this.obj.a = false;
                    this.obj.b = true;
                }
                
            }
        }
    })
    </script>
</body>

    <div id="root">
        <h2>遍历数组</h2>
        <button @click.once="add">添加1人</button>
        <ul>
            <li v-for="(p,index) of person" :key="p.id">
                {{p.name}}-{{p.age}}
                <input type="text">
            </li>
            
        </ul>
    </div>
    <script>

    let vm = new Vue({
        el:'#root',
        data(){
            return{
                person:[
                    {id:'001',name:'a',age:18},
                    {id:'002',name:'b',age:33},
                    {id:'003',name:'c',age:22},
                ]
            }
        },
        methods:{
            add(){
                this.person.unshift({id:"004",name:'d',age:24})
            }
        }
        
    })
    </script>
</body>
</html>
